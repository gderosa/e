    \hypersetup{
        linkcolor = DarkRed,
        citecolor = Green,
        urlcolor  = DarkBlue
    }

\begin{document}
    \frontmatter
        % Metadata
        \title{The Title}
        \author{The Author}

        % Print front matter meta
        \maketitle
        {
            \hypersetup{
              linkcolor = .
            }

            \tableofcontents
        }

        % Unnumbered chapters (Abstract, Acknowledgments, etc.)
        \input{tex/pre/abs.tex}
        \input{tex/pre/ack.tex}

    \mainmatter
        % Proper chapters
        \input{tex/ch/intro}

        % Appendices
        \appendix
        \input{tex/ap/a}

    \backmatter
    % Biblio
\end{document}
